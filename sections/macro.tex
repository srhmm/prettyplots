 \subsection*{Commands for Line Plots}
 \hypertarget{addlines}{}
 \begin{function}{\tbs addlines}
 	\docarg{data.tsv}{s}   
 	\docarg{addplot options}{l}  
 	\docarg{table options}{l}  
 	\\ adds all line plots from a given file, from columns \texttt{(x, y1)}, \ldots, \texttt{(x, yn)}  .  
 \end{function}

 \hypertarget{addconflines}{}
 \begin{function}{\tbs addconflines}
 	\docarg{data.tsv}{s}  \docarg{addplot options}{l} 	\docarg{table options}{l}  
 	\\ adds line plots and confidence regions from a given file, from columns \texttt{(x, y1, y1\_c)}, \ldots, \texttt{(x, yn, yn\_c)}.
 \end{function}
 
 \hypertarget{grouplines}{}
 \begin{function}{\tbs grouplines}
 	\docarg{data.tsv}{s}  \docarg{addplot options}{l}  	\docarg{table options}{l}  
 	\\ adds line plots to a group plot.
 \end{function}
  \hypertarget{groupconflines}{}
 \begin{function}{\tbs groupconflines}
 	\docarg{data.tsv}{s}  \docarg{addplot options}{l}  	\docarg{table options}{l}  
 	\\ adds line plots to a group plot.
 \end{function}
 
 
 \begin{arglist}
 	\argitem{data.tsv}{s}{m}{data file.}   
% 	\argditem{columns}{l}{o}{column names to plot.}{\texttt{y1, ..., yn}.} 
 	\argeitem{addplot options}{l}{o}{style for adding plots, such as color, line-, and markers style.
 	}{black, smooth, line width=1.2pt, mark=*, mark options=\{fill opacity=0.5,mark size=1pt\}} 
 	\argeitem{table options}{l}{o}{keys to \texttt{table}. }{col sep=comma} 
 \end{arglist}

\subsection{Scatter}
\index{Plot Types!scatter} 
Use this plot type if you want color and style of markers to depend on a given class label. 

\hypertarget{addscatter}{}
\begin{function}{\tbs addscatter}
	\docarg{data.tsv}{s}  
	\docarg{plot options}{l}  
	\docarg{table options}{l}  
	\\ creates a scatter plot from a table file.
\end{function}


\begin{arglist}
	\argitem{data.tsv}{s}{m}{data file.} 
	\argeitem{plot options}{l}{o}{style for adding plots, such as colors per scatter class. }{\texttt{pr-scatter}}
	\argditem{table options}{l}{o}{options for data table, such as column names of \texttt{x,y,meta}, where \texttt{meta} is a numeric class label for coloring. }{\texttt{x=x, y=y, meta=label}} 
\end{arglist}


\begin{keylist}{Plot Options} 
	\keyitem{pr-scatter}{default colors}  
	\keyitem{pr-scatter-filled}{mark fill variant}
	\keyitem{pr-scatter-complementary}{mark color variant}
\end{keylist}


\begin{keylist}{Table Options} 
	\keyitem{meta=label}{change name of the column with class labels here, default is \texttt{label}.  }   
\end{keylist}
