\subsection{Error Line}  
\index{Plot Types!line!with confidence regions}
If you want to plot confidence regions around your line, include them in your dataset as follows,

\begin{figure}[h!]
	\begin{minipage}{0.65\textwidth}
		\begin{mintedtxtbox}[/data/line\_conf.tsv]

			\end{minted}
		\end{mintedtxtbox}

\end{figure}

Note that the upper and lower confidence lines are each added using addplot, where \texttt{forget plot} makes sure that these lines are disregarded in the legend and cycle list. The last \texttt{fill-between} line is added without  \texttt{forget plot} as we  define its color in the cycle list. If you use a legend, skip these plots as follows.
\begin{figure}[h!] 
	\begin{minipage}{0.65\textwidth} 
		\begin{tcolorbox}
			\begin{minted}{TeX}   
\begin{axis}[
    pretty line,
    pretty fill legend, 
    cycle list name = pr-colors-conf,
    % double ,, skips the confidence regions in the legend:
    legend entries = {Method 1,,Method 2,,Method 3},
]  % ...
\end{axis}   
			\end{minted}
		\end{tcolorbox}
	\end{minipage}
	\begin{minipage}{0.3\textwidth}   
		\begin{tikzpicture} 
				\begin{axis}[
					pretty line, 
					pretty fill legend,
					ymin=0, ymax=10,  width=\textwidth, 
					cycle list name = pr-colors-conf,
					% double ,, skips the confidence regions in the legend:
					legend entries = {Method 1,,Method 2,,Method 3},
					legend style={at={(1,1)}, anchor=north west}, 
					]  
	\foreach \r in {1,2,3}   {
	\addplot table[x=x, y=y\r] {data/line_conf.tsv}; 
	\addplot[forget plot, draw=none, name path=low] 
	table [x=x, y expr=\thisrow{y\r} - \thisrow{y\r_c}]
	{data/line_conf.tsv};
	\addplot[forget plot, draw=none, name path=up] 
	table [x=x, y expr= \thisrow{y\r} + \thisrow{y\r_c}]
	{data/line_conf.tsv};
	\addplot fill between[of=low and up];
}  
				\end{axis}  
			\end{tikzpicture} 
		%\lines{data/line.tsv}[pretty grid, cycle list name=pr-marks]
		\hfill	
	\end{minipage} 	 
\end{figure}

To show error bars instead, use
	\begin{figure}[h!] 
		\begin{minipage}{0.65\textwidth} 
			\hypertarget{adderrorlines}{}
			\begin{tcolorbox}
				\begin{minted}{TeX} 
\begin{axis}[pretty line]
    \foreach \r in {1,...,3}{ 
        \addplot+[error bars/.cd, y explicit, y dir=both] 
        table[ x=x, y=y\r, 
        y error plus expr=\thisrow{y\r_c},
        y error minus expr=\thisrow{y\r_c}, 
        ] {data/line_conf.tsv};
}  
\end{axis}
				\end{minted}
			\end{tcolorbox}
		\end{minipage}
		\begin{minipage}{0.3\textwidth} 
			\begin{tikzpicture}
				\begin{axis}[
					pretty line,  width=\textwidth
					]
					\foreach \col in {1,...,3}{ 
						\addplot+[ 
						error bars/.cd,
						y explicit,
						y dir=both] table[
						x=x, y=y\col, 
						y error plus expr=\thisrow{y\col_c},
						y error minus expr=\thisrow{y\col_c}, 
						] {data/line_conf.tsv};
					}  
					%\adderrorlines{data/line_conf.tsv}
				\end{axis}
