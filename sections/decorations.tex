The following examples are inspired from \cite{knaflic2015storytelling}.

\index{Plot Types!line!simplified}
\index{Labels!near lines}
\begin{figure}[h!]
	\begin{minipage}{0.65\textwidth} 
		\begin{tcolorbox}
			\begin{minted}{TeX} 
\begin{axis}[
    pretty line,    
    ytick = {0, 5, 10},
    x tick label style = {pr-color-gray5}, 
    y tick label style = {pr-color-gray5},  
    xtick style = {pr-color-gray4, thick},
    ytick style = {pr-color-gray4, thick},
    x axis line style = {pr-color-gray4, very thick}, 
    y axis line style = {pr-color-gray4, very thick} 
    ] 
    \addplot+[line width=1.5pt] 
    table[x=x, y expr=\thisrow{y1}+1] 
        {data/line.tsv} 
        node[pos=1, right] (n) {\small \textbf{Ours}};    
    \pgfplotsinvokeforeach{2,...,3}{
    \addplot+[pr-color-gray4] table[x=x, y=y#1] 
        {data/line.tsv}
        node[pos=1, right] (n) {\small Method #1};   
    }
\end{axis}
\end{minted}
		\end{tcolorbox}
	\end{minipage}
	\begin{minipage}{0.3\textwidth} 
		\def\methods{ourmethod, goodmethod, badmethod, uglymethod}
		\def\file{data/box.tsv} 
		\begin{tikzpicture} 
			\begin{axis}[
				pretty line,    
				xticklabel style = {pr-color-gray5}, 
				yticklabel style = {pr-color-gray5},  
				xtick style = {pr-color-gray4, thick},
				x axis line style = {pr-color-gray4, very thick}, 
				ytick style = {pr-color-gray4, thick},
				y axis line style = {pr-color-gray4, very thick}, 
				ytick = {0, 5, 10},
				ymin=0, width=\textwidth, 
				] 
				\addplot+[line width=1.5pt] table[x=x, y expr=\thisrow{y1}+1] {data/line.tsv} node[pos=1, right] (n) {\small \textbf{ Ours}};  
				\pgfplotsinvokeforeach{2,...,3}{
					\addplot+[pr-color-gray4] table[x=x, y=y#1] {data/line.tsv} node[pos=1, right] (n) {\small Method #1};   
				}
			\end{axis}
		\end{tikzpicture}
	\end{minipage} 
\end{figure}
